## 함수 (func)

### 함수 선언
```go
package main

import "fmt"

func main() {
  fmt.Println(hello1(1))

  fmt.Println(hello2(2, 3))

  var1, var2 := hello3(3, 4)
  fmt.Println(var1, var2)

  var3, _, var4, _ := hello4(1, 2)
  fmt.Println(var3, var4)
}

// 함수를 main 이후에 선언해도 컴파일 오류가 생기지 않는다.
func hello1(a int) int {
	return a * 2
}

// return 값의 이름을 지정해놓으면 return 만 써도 r을 리턴한다.
func hello2(a int, b int) (r int) {
	r = a + b
	return
}

// return을 두개로 둘 수도 있다.
func hello3(a int, b int) (r int, rr int) {
	r = a + b
	rr = a * b
	return
}

// return을 두개로 두고, 이름을 붙이지 않을때.
func hello3_1(a int, b int) (int, int) {
  return a + b, a * b
}

// 이럴때 _를 써서 선택적으로 return을 받을 수 있다.
func hello4(a int, b int) (r int, rr int, rrr int, rrrr int) {
	r = a + b
	rr = a * b
	rrr = a + a + a + b
	rrrr = a*2 + b
	return
}
```

### 함수 파라미터 전달방식
* 파이썬을 하다보니까 차라리 포인터가 반갑다.
```go
package main

import "fmt"

func main() {
  val1, val2 := 1, 5

  hello1(val1)
  fmt.Println(val1)
  //1 다음 2


  fmt.Println(val2)
  hello2(&val2)
  fmt.Println(val2)
  //5 다음 6, 그리고 6
}

// call by value
func hello1(a int) {
	a += 1
	fmt.Println(a)
}

// call by reference
func hello2(a *int) {
	*a += 1
	fmt.Println(*a)
}
```

### 가변 인자
```go
package main

import "fmt"

func main() {
	hello1(23, 54, 23, 12, 32, 54)
  // [23 54 23 12 32 54]

	hello2(23, 54, 23, 12, 32, 54)
  /* 0-23
   * 1-54
   * 2-23
   * 3-12
   * 4-32
   * 5-54
   */
}

// ...자료형으로 받으면 가변 길이의 인자를 받을 수 있다.
func hello1(a ...int) {
	fmt.Println(a)
}

// 여기서 또 range를 쓰면 이게 또 별미임
func hello2(a ...int) {
	for val1, val2 := range a {
		fmt.Print(val1)
		fmt.Print("-")
		fmt.Println(val2)
	}
}
```
